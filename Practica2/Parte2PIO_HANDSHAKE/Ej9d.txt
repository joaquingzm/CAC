        SUB BX,CX

        MOV AL,[BX]
        OUT PIO+1,AL

        OUT PIO,AL

        OUT PIO,AL

        IN AL,PIO
        AND AL,00000001B

        INC BX
        DEC CL
        JNZ BUCLE_AUN_CHARS
        RET

        OUT PIO+2,AL
        ;SETEO DE PB

        OUT PIO+3,AL

        MOV CX,0
