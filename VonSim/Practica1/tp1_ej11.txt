ORG 1000H
	DIVIDENDO DB 10
	DIVISOR DB 2
	RES DB ?
ORG 3000H
DIV:
	PUSH AX
	PUSH BX
	PUSH DX
	MOV DL,0
	MOV BX,SP
	ADD BX,12	;BX = PILA OFFSET 
	PUSH BX
	SUB BX,2	;BX = PILA DIVIDENDO
	MOV AL,[BX]
	SUB BX,2
	MOV AH,[BX]
	POP BX
	MOV BX, [BX]
FOR: 
	CMP AL,AH
	JS FIN
	SUB AL,AH
	INC DL
	JMP FOR 
FIN:
	MOV [BX],DL
	POP DX
	POP BX
	POP AX
	RET
ORG 2000H
	MOV AX, OFFSET RES
	PUSH AX
	MOV AX,DIVIDENDO
	PUSH AX
	MOV AX, DIVISOR
	PUSH AX
	CALL DIV
	POP AX
	POP AX
END
