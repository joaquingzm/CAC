ORG 1000H
  NUM1 DW 5
  NUM2 DW 3
ORG 3000H
 
SWAP:
  PUSH AX
  PUSH BX
  PUSH CX
  PUSH DX
  MOV BX,SP
  ADD BX,12
  MOV BX,[BX]
  PUSH BX
  MOV AX,[BX]	;AX=5,NUM1
  MOV BX,SP
  ADD BX,12
  MOV BX,[BX]
  MOV CX,[BX]	;CX=3,NUM2

  XOR AX,CX
  XOR CX,AX
  XOR AX,CX

  MOV [BX],CX
  POP BX ;DIR DE NUM2
  MOV [BX],AX

  POP DX
  POP CX
  POP BX
  POP AX
RET
ORG 2000H
  MOV BX,OFFSET NUM1
  PUSH BX
  MOV BX,OFFSET NUM2
  PUSH BX
  CALL SWAP
  POP BX
  POP BX
END
