ORG 1000H
NUM DB 11001001B
N DB 3
ORG 3000H
ROTIZQ:
  ADD AH,AH
  JNC NO_C
  INC AH
NO_C:
  MOV [BX],AH
  RET
ROTIZQ_N:
  CALL ROTIZQ
  DEC AL
  JNZ ROTIZQ_N
FIN:
  RET
ROTDER_N:
  MOV CH,8
  SUB CH,AL
FOR:
  CALL ROTIZQ
  DEC CH
  JNZ FOR
  RET   
ORG 2000H
  MOV AH,NUM
  MOV BX,OFFSET NUM
  MOV AL,N
  CALL ROTIZQ_N 
  HLT
END
