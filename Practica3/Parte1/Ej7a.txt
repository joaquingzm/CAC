        HANDSHAKE EQU 40H
        POS_GESTOR_H EQU 4
        PIC EQU 20H
        ORG 16
        DW GESTOR_H

        ORG 3000H
GESTOR_H:
        DEC CL
        IN AL,PIC+1
        XOR AL,00000100B
        OUT PIC+1,AL
SEGUIR: MOV AL,[BX]
        INC BX
        OUT HANDSHAKE,AL
        MOV AL,20H
        OUT PIC,AL
        IRET
        ORG 1000H
        STRING DB "UNIVERSIDAD NACIONAL DE LA PLATA"
        FIN DB ?

        ORG 2000H
        CLI
        MOV AL,POS_GESTOR_H
        OUT PIC+6,AL ;ASIGNO POS DEL VECTOR DE INTERRUPCIONES A INT2
        MOV AL,11111011B
        OUT PIC+1,AL ;HABILITO INT2
        MOV AL,10000000B
        OUT HANDSHAKE+1,AL ;HABILITO INTERRUPCIONES DESDE HANDSHAKE
        MOV CL,OFFSET FIN - OFFSET STRING
        MOV BX,OFFSET STRING
        STI
FOR:    CMP CL,0
        JNZ FOR
        HLT
        END
